Load prepared data from the previous step "Prepare Data For Modeling"

\begin{alltt}
\hlkwd{rm}\hlstd{(}\hlkwc{list}\hlstd{=}\hlkwd{ls}\hlstd{())}   \hlcom{#clear memory}

\end{alltt}
\end{kframe}
\end{knitrout}

Split the train data based on simple bootstrap resampling into a series of train and test sets
\begin{knitrout}
\definecolor{shadecolor}{rgb}{0.969, 0.969, 0.969}\color{fgcolor}\begin{kframe}
\begin{alltt}
\hlkwd{library}\hlstd{(caret)}
\hlkwd{set.seed}\hlstd{(}\hlnum{998}\hlstd{)}
\hlstd{PERCENTAGE} \hlkwb{<-} \hlnum{0.07}
\hlstd{inTraining} \hlkwb{<-} \hlkwd{createDataPartition}\hlstd{(rfModelData}\hlopt{$}\hlstd{is_delayed,} \hlkwc{times}\hlstd{=}\hlnum{1}\hlstd{,} \hlkwc{p} \hlstd{= PERCENTAGE,} \hlkwc{list} \hlstd{=} \hlnum{FALSE}\hlstd{)}
\hlkwd{length}\hlstd{(inTraining)}
\end{alltt}
\begin{verbatim}
## [1] 516206
\end{verbatim}
\begin{alltt}
\hlstd{training} \hlkwb{<-} \hlstd{rfModelData[inTraining,]}
\hlcom{# testing  <- rfModelData[-inTraining,]}
\end{alltt}
\end{kframe}
\end{knitrout}

Estimate a random forest using 7\% of the data - without crossvalidation:
\begin{knitrout}
\definecolor{shadecolor}{rgb}{0.969, 0.969, 0.969}\color{fgcolor}\begin{kframe}
\begin{alltt}
\hlkwd{library}\hlstd{(randomForest)}
\hlstd{delayRf} \hlkwb{<-} \hlkwd{randomForest}\hlstd{(is_delayed} \hlopt{~} \hlstd{.} \hlopt{-} \hlstd{id,} \hlkwc{data}\hlstd{=rfModelData,} \hlkwc{subset}\hlstd{=inTraining,}
                        \hlkwc{importance}\hlstd{=}\hlnum{TRUE}\hlstd{,} \hlkwc{proximity}\hlstd{=}\hlnum{FALSE}\hlstd{)}
\end{alltt}
\end{kframe}
\end{knitrout}
Note: On a Macbook with 16GB RAM it takes 6 minutes for sample size of 100.000 and 30 minutes fortraining sample size of 5\% (about 360.000 obs).
