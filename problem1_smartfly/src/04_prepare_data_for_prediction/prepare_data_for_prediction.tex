\section{Remove unknown factor levels}

The comparison of historic and scheduled flight data furthermore showed that there are two airlines in the scheduled data that are not listed in the historic data. In order to be able to predict using the estimated random forest from the historic data I need to exclude the observations listed for these two airlines:
\begin{knitrout}
\definecolor{shadecolor}{rgb}{0.969, 0.969, 0.969}\color{fgcolor}\begin{kframe}
\begin{alltt}
\hlstd{removeNames} \hlkwb{<-} \hlkwd{c}\hlstd{(}\hlstr{"HA"}\hlstd{,}\hlstr{"OH"}\hlstd{)}
\hlstd{removeIdxs} \hlkwb{<-} \hlkwd{union}\hlstd{(}\hlkwd{which}\hlstd{(rfPredictData}\hlopt{$}\hlstd{airline}\hlopt{==}\hlstr{"HA"}\hlstd{),} \hlkwd{which}\hlstd{(rfPredictData}\hlopt{$}\hlstd{airline}\hlopt{==}\hlstr{"OH"}\hlstd{))}
\hlkwd{length}\hlstd{(removeIdxs)}
\end{alltt}
\begin{verbatim}
## [1] 33398
\end{verbatim}
\begin{alltt}
\hlstd{rfPredictData} \hlkwb{<-} \hlstd{rfPredictData[}\hlopt{-}\hlstd{removeIdxs,]}
\hlstd{lvlIdx} \hlkwb{<-} \hlkwd{c}\hlstd{(}\hlkwd{which}\hlstd{(}\hlkwd{levels}\hlstd{(rfPredictData}\hlopt{$}\hlstd{airline)}\hlopt{==}\hlstr{"HA"}\hlstd{),} \hlkwd{which}\hlstd{(}\hlkwd{levels}\hlstd{(rfPredictData}\hlopt{$}\hlstd{airline)}\hlopt{==}\hlstr{"OH"}\hlstd{))}
\hlkwd{levels}\hlstd{(rfPredictData}\hlopt{$}\hlstd{airline)[lvlIdx]} \hlkwb{<-} \hlnum{NA}
\end{alltt}
\end{kframe}
\end{knitrout}

\section{Analyse \& deal with missing values}

\begin{knitrout}
\definecolor{shadecolor}{rgb}{0.969, 0.969, 0.969}\color{fgcolor}\begin{kframe}
\begin{alltt}
\hlstd{nbRows} \hlkwb{<-} \hlkwd{dim}\hlstd{(rfPredictData)[}\hlnum{1}\hlstd{]}
\hlstd{rowHasNa} \hlkwb{<-} \hlkwd{apply}\hlstd{(rfPredictData,} \hlkwc{MARGIN}\hlstd{=}\hlnum{1}\hlstd{,} \hlkwc{FUN}\hlstd{=}\hlkwa{function}\hlstd{(}\hlkwc{row}\hlstd{)\{} \hlkwd{any}\hlstd{(}\hlkwd{is.na}\hlstd{(row)) \})}
\hlstd{nbRowsWithNa} \hlkwb{<-} \hlkwd{sum}\hlstd{(rowHasNa)}
\hlstd{nbRowsLeft} \hlkwb{<-} \hlstd{nbRows} \hlopt{-} \hlstd{nbRowsWithNa}
\hlcom{# proportion of NA rows:}
\hlstd{nbRowsWithNa} \hlopt{/} \hlstd{nbRows}
\end{alltt}
\begin{verbatim}
## [1] 0
\end{verbatim}
\end{kframe}
\end{knitrout}
